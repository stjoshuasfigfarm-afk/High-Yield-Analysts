ASSET & COMMODITY DATA --- const assets = [ { id: 'AAPL', n: 'Apple', c: [37.33, -122.03], color: '#10b981', type: 'Tech' }, { id: 'NVDA', n: 'NVIDIA', c: [37.37, -121.97], color: '#22ab94', type: 'Tech' }, { id: 'WMT', n: 'Walmart', c: [36.37, -94.20], color: '#3b82f6', type: 'Retail' }, { id: 'XOM', n: 'ExxonMobil', c: [32.89, -96.94], color: '#f97316', type: 'Energy' } ]; // Map your new commodities to vessel types const vesselClasses = { 'TANKER': { color: '#f97316', icon: 'M12 2L22 20L12 17L2 20L12 2Z', cargo: ['Crude Oil', 'LNG', 'Heating Oil'] }, 'BULK': { color: '#eab308', icon: 'M3 3h18v18H3V3z', cargo: ['Lithium', 'Copper', 'Iron Ore', 'Soybeans'] }, 'CONTAINER': { color: '#3b82f6', icon: 'M4 6h16v12H4z', cargo: ['Processed Silicon', 'Retail Electronics', 'Auto Parts'] } }; // --- 3. UPGRADED MARITIME LOGIC --- let activeVessels = {}; const getVesselIcon = (heading, typeInfo) => { return L.divIcon({ className: 'vessel-icon', html: ``, iconSize: [14, 14], iconAnchor: [7, 7] }); }; function updateVessel(mmsi, lat, lon, heading, vClass) { const typeInfo = vesselClasses[vClass]; const randomCargo = typeInfo.cargo[Math.floor(Math.random() * typeInfo.cargo.length)]; if (activeVessels[mmsi] && activeVessels[mmsi].marker) { activeVessels[mmsi].marker.setLatLng([lat, lon]); activeVessels[mmsi].marker.setIcon(getVesselIcon(heading, typeInfo)); } else { const marker = L.marker([lat, lon], { icon: getVesselIcon(heading, typeInfo) }).addTo(maritimeLayer); marker.bindTooltip(` ${vClass} CLASS
Est. Cargo: ${randomCargo}
Heading: ${Math.round(heading)}° `, { direction: 'top', className: 'leaflet-tooltip-own', offset: [0, -5] }); activeVessels[mmsi] = { marker, vClass }; } } // ========================================== // ADVANCED SIMULATION ENGINE (Trade Routes) // ========================================== function startSimulationFallback() { // Defined strategic trade corridors based on your commodities const routes = [ { lat: 24, lon: -90, head: 120, class: 'TANKER' }, // US Gulf Coast (Crude/LNG) { lat: -20, lon: -70, head: 320, class: 'BULK' }, // Chile/Peru Coast (Lithium/Copper) { lat: 15, lon: 115, head: 60, class: 'CONTAINER' }, // South China Sea (Tech/Retail) { lat: 50, lon: -10, head: 220, class: 'TANKER' } // North Sea (Brent Crude) ]; routes.forEach((r, i) => { for(let j=0; j<8; j++) { let mmsi = `SIM_${i}_${j}`; let lat = r.lat + (Math.random() * 8 - 4); let lon = r.lon + (Math.random() * 8 - 4); activeVessels[mmsi] = { lat, lon, head: r.head, class: r.class, speed: 0.03 + Math.random()*0.04 }; updateVessel(mmsi, lat, lon, r.head, r.class); } }); setInterval(() => { for(let mmsi in activeVessels) { let v = activeVessels[mmsi]; if(!v.speed) continue; let rad = (v.head - 90) * (Math.PI / 180); v.lat += Math.sin(rad) * v.speed; v.lon += Math.cos(rad) * v.speed; v.head += (Math.random() * 2 - 1); // Drift updateVessel(mmsi, v.lat, v.lon, v.head, v.class); } }, 1000); } // --- 4. UI INITIALIZATION --- function init() { // Plot HQs assets.forEach(e => { const marker = L.circleMarker(e.c, { radius: 5, color: '#fff', weight: 1, fillColor: e.color, fillOpacity: 0.8 }).addTo(hqLayer); marker.bindTooltip(`${e.n} HQ`, { direction: 'top', className: 'leaflet-tooltip-own', offset: [0, -5] }); }); startSimulationFallback(); } window.onload = init;
